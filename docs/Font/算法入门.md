#### 多个数据比较
```javascript
//对长数据进行拆分
let num = [5,3,4,5,3,2]
//已经实现了对两个数据进行比较的方法，多个数据思路如下
let num1 = [5,[3,4,5,3,2]]
let num2 = [3,[4,5,3,2]]
let num3 = [4,[5,3,2]]
let num4 = [5,[3,2]]
//将一个长数据变成两两比较的形式
//递归需要一个终止条件，否则会一直调用自己
function min(num){
	if(num.length>2){
		//当数组长度大于2时，则将数组第一项与后面的所有值，当做一个两项数组
		return min(num[0],min(num.slice(1)))  //slice(1)将第一项除去
	}else{
		return Math.min.apply(null,num)
	}
}
```
